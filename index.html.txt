<script>
    // 自動填入當前時間到假單輸入框
    setInterval(() => {
        let now = new Date();
        document.getElementById("lSubTime").value = now.toLocaleString('zh-TW', { hour12: false });
    }, 1000);

    // --- 分頁切換功能 ---
    function openTab(tabName) {
        var i, tabcontent, tabbtns;
        tabcontent = document.getElementsByClassName("tab-content");
        for (i = 0; i < tabcontent.length; i++) {
            tabcontent[i].classList.remove("active");
        }
        tabbtns = document.getElementsByClassName("tab-btn");
        for (i = 0; i < tabbtns.length; i++) {
            tabbtns[i].classList.remove("active");
        }
        document.getElementById(tabName).classList.add("active");
        event.currentTarget.classList.add("active");
    }

    // --- 成員管理功能 ---
    function addMember() {
        let id = document.getElementById("mID").value;
        let job = document.getElementById("mJob").value;
        let force = document.getElementById("mForce").value;
        let note = document.getElementById("mNote").value;
        if(id === "") return alert("請輸入 ID");

        let tbody = document.getElementById("guildTable").querySelector("tbody");
        let row = tbody.insertRow();
        row.innerHTML = `
            <td contenteditable="true" class="editable">${id}</td>
            <td contenteditable="true" class="editable">${job}</td>
            <td class="center-col" onclick="toggleForce(this)"><span class="${force==='是'?'force-yes':'force-no'}">${force}</span></td>
            <td contenteditable="true" class="editable">${note}</td>
            <td class="center-col"><button class="ro-btn btn-delete" onclick="deleteRow(this)">移除</button></td>
        `;
        document.getElementById("mID").value = "";
    }

    // --- 假單管理功能 ---
    function addLeave() {
        let subTime = document.getElementById("lSubTime").value;
        let name = document.getElementById("lName").value;
        let time = document.getElementById("lTime").value;
        let reason = document.getElementById("lReason").value;
        if(name === "") return alert("請輸入遊戲名稱");

        let tbody = document.getElementById("leaveTable").querySelector("tbody");
        let row = tbody.insertRow();
        row.innerHTML = `
            <td>${subTime}</td>
            <td contenteditable="true" class="editable">${name}</td>
            <td contenteditable="true" class="editable">${time}</td>
            <td contenteditable="true" class="editable">${reason}</td>
            <td class="center-col"><button class="ro-btn btn-delete" onclick="deleteRow(this)">刪除</button></td>
        `;
        document.getElementById("lName").value = "";
        document.getElementById("lReason").value = "";
    }

    // 共用功能
    function deleteRow(btn) {
        btn.closest("tr").remove();
    }

    function toggleForce(td) {
        let span = td.querySelector("span");
        if (span.innerText === "是") {
            span.innerText = "否"; span.className = "force-no";
        } else {
            span.innerText = "是"; span.className = "force-yes";
        }
    }

    function sortTable(tableId, n) {
        var table, rows, switching, i, x, y, shouldSwitch, dir, switchcount = 0;
        table = document.getElementById(tableId);
        switching = true;
        dir = "asc"; 
        while (switching) {
            switching = false;
            rows = table.rows;
            for (i = 1; i < (rows.length - 1); i++) {
                shouldSwitch = false;
                x = rows[i].getElementsByTagName("TD")[n];
                y = rows[i + 1].getElementsByTagName("TD")[n];
                let xVal = x.innerText.toLowerCase();
                let yVal = y.innerText.toLowerCase();
                if (dir == "asc") {
                    if (xVal > yVal) { shouldSwitch = true; break; }
                } else if (dir == "desc") {
                    if (xVal < yVal) { shouldSwitch = true; break; }
                }
            }
            if (shouldSwitch) {
                rows[i].parentNode.insertBefore(rows[i + 1], rows[i]);
                switching = true; switchcount++;
            } else {
                if (switchcount == 0 && dir == "asc") { dir = "desc"; switching = true; }
            }
        }
    }
